<objective>
Add `QueueURLTemplate` per-table routing to `SQSSinkConsumer`. When `queue-url-template` is configured, `Deliver()` evaluates the Go template per-message and routes to the resolved SQS FIFO queue URL. A lazy `validatedQueues map[string]bool` pool (protected by `sync.RWMutex`) ensures FIFO validation runs once per unique queue URL, not per message.

Purpose: Close the CFG-02 structural gap for SQS — currently all CDC events route to a single fixed `QueueURL`; this enables per-table routing across multiple FIFO queues.
Output: Updated `SQSSinkConfig` config struct + updated `SQSSinkConsumer` with template routing + pool.
</objective>

```go
func (c *SQSSinkConsumer) getOrValidateQueue(ctx context.Context, queueURL string) error {
    c.mu.RLock()
    ok := c.validatedQueues[queueURL]
    c.mu.RUnlock()
    if ok {
        return nil
    }
    c.mu.Lock()
    defer c.mu.Unlock()
    if c.validatedQueues[queueURL] {
        return nil
    }
    out, err := c.client.GetQueueAttributes(ctx, &sqs.GetQueueAttributesInput{
        QueueUrl:       aws.String(queueURL),
        AttributeNames: []types.QueueAttributeName{types.QueueAttributeNameFifoQueue},
    })
    if err != nil {
        return fmt.Errorf("sqs sink: get queue attributes for %q: %w", queueURL, err)
    }
    if out.Attributes[string(types.QueueAttributeNameFifoQueue)] != "true" {
        return fmt.Errorf("sqs sink: %q is not a FIFO queue — SQS sink requires a .fifo queue URL", queueURL)
    }
    c.validatedQueues[queueURL] = true
    return nil
}
```
